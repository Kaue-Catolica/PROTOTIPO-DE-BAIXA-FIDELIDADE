Introdução

As estruturas condicionais permitem que seu programa tome decisões com base em condições. As principais são if, if-else, switch. Além disso, temos estruturas de repetição, como while e for, que permitem executar blocos de código várias vezes.

Estrutura If

A estrutura if verifica se uma condição é verdadeira e, caso seja, executa um bloco de código. Caso contrário, esse bloco é ignorado.


#include <stdio.h>

int main() {
    int idade = 18;
    if (idade >= 18) {
        printf("Você é maior de idade.");
    }
    return 0;
}

Estrutura If-Else

O if-else adiciona uma alternativa. Se a condição no if não for verdadeira, o bloco de código no else será executado.


#include <stdio.h>

int main() {
    int idade = 16;
    if (idade >= 18) {
        printf("Você é maior de idade.");
    } else {
        printf("Você é menor de idade.");
    }
    return 0;
}

Estrutura Switch

A estrutura switch permite testar uma variável contra várias condições, executando blocos de código diferentes com base no valor da variável.


#include <stdio.h>

int main() {
    int dia = 3;

    switch(dia) {
        case 1:
            printf("Domingo");
            break;
        case 2:
            printf("Segunda-feira");
            break;
        case 3:
            printf("Terça-feira");
            break;
        default:
            printf("Dia inválido");
    }
    return 0;
}

Estruturas de Repetição

As estruturas de repetição while e for são usadas para executar um bloco de código várias vezes enquanto uma condição for verdadeira.

Exemplo de While:


#include <stdio.h>

int main() {
    int i = 0;
    while (i < 5) {
        printf("%d\n", i);
        i++;
    }
    return 0;
}

Exemplo de For:


#include <stdio.h>

int main() {
    for (int i = 0; i < 5; i++) {
        printf("%d\n", i);
    }
    return 0;
}

Exercícios

Pratique o uso de estruturas condicionais com os exemplos a seguir:

1. Verifique se um número é par ou ímpar:


#include <stdio.h>

int main() {
    int numero;
    printf("Digite um número: ");
    scanf("%d", &numero);

    if (numero % 2 == 0) {
        printf("O número é par.");
    } else {
        printf("O número é ímpar.");
    }
    return 0;
}

2. Crie um programa que verifica o dia da semana com switch:


#include <stdio.h>

int main() {
    int dia;
    printf("Digite um número (1-7) para o dia da semana: ");
    scanf("%d", &dia);

    switch(dia) {
        case 1:
            printf("Domingo");
            break;
        case 2:
            printf("Segunda-feira");
            break;
        case 3:
            printf("Terça-feira");
            break;
        case 4:
            printf("Quarta-feira");
            break;
        case 5:
            printf("Quinta-feira");
            break;
        case 6:
            printf("Sexta-feira");
            break;
        case 7:
            printf("Sábado");
            break;
        default:
            printf("Dia inválido");
    }
    return 0;
}

Desafios

Testes seu conhecimento com esses desafios:

1. Calcule o desconto em uma compra baseado no valor:


#include <stdio.h>

int main() {
    float valor;
    printf("Digite o valor da compra: ");
    scanf("%f", &valor);

    if (valor >= 500) {
        printf("Desconto de 20%%.");
    } else if (valor >= 300) {
        printf("Desconto de 15%%.");
    } else {
        printf("Desconto de 10%%.");
    }

    return 0;
}

2. Crie um jogo onde o usuário deve adivinhar um número entre 1 e 10:


#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    int numero, tentativa;
    srand(time(NULL));
    numero = rand() % 10 + 1;

    printf("Adivinhe o número (entre 1 e 10): ");
    do {
        scanf("%d", &tentativa);
        if (tentativa > numero) {
            printf("Muito alto! Tente novamente: ");
        } else if (tentativa < numero) {
            printf("Muito baixo! Tente novamente: ");
        }
    } while (tentativa != numero);

    printf("Parabéns! Você adivinhou o número!");
    return 0;
}
